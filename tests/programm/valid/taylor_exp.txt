derivaten( f, var, 0 ) := f;
derivaten( f, var, 1 ) := derivate( Force(f), Force(var) );
derivaten( f, var, n ) := derivate( Force(derivaten( f, var, Force(n-1) )), Force(var) );

taylor( f, var, a, 0 ) := f;
taylor( f, var, a, n ) := taylor( f, var, a, n- 1 )
                       + (derivaten(f, var, Force(n)))( a ) / n! * (x - a) ^ n;
