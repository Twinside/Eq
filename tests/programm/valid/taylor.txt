-- derivaten( function, var, order )
derivaten( f, var, 0 ) := f;
derivaten( f, var, 1 ) := derivate( Force(f), Force(var) );
derivaten( f, var, n ) := derivate( Force(derivaten( f, var, n-1 )), Force(var) );

sinus( z ) := sin( z );

taylorn( f, var, a, n, n ) := (derivaten(f, var, n))( a ) / n! * (x - a) ^ n;
taylorn( f, var, a, max, 0 ) := f + taylorn( f, var, a, max, 1 );
taylorn( f, var, a, max, n ) := (derivaten(f, var, n))( a ) / n! * (x - a) ^ n
                             + taylorn( f, var, a, max, n + 1 );

-- taylor( function, derivation var, onVar, order )
taylor( f, var, a, 0 ) := f;
taylor( f, var, a, n ) := taylorn( f, var, a, n, 0 );
